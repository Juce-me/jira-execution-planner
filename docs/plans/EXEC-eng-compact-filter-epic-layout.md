# ENG Compact Filter And Epic Layout Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Make the ENG Catch Up filter cards and epic/story blocks more compact while keeping all visible information readable and preserving EPM issue-board layout.

> **Execution update 2026-06-09:** Implemented for the desktop ENG Catch Up view. Focused ENG compact-layout, source-guard, build, EPM issue-board, ENG visual, sticky smoke, and EPM portfolio-header visual checks pass. Narrow viewport checks are guardrails for overflow and EPM compatibility only; full mobile layout polish remains out of scope for this commit.

**Architecture:** Use CSS-only scoped overrides. ENG filter-card changes must live under `.filters-strip ...`; ENG epic/story density changes must live under `.task-list:not(.epm-issue-board) ...` so shared EPM `IssueCard` and rollup markup keeps its current spacing. Add source guards before CSS changes so accidental unscoped selectors fail fast.

**Tech Stack:** React 19, existing JSX markup, CSS partials bundled through `frontend/src/styles/dashboard.css`, Node `node:test`, Playwright UI specs.

## Scope Guards

CSS scope guard: compact ENG story/epic overrides must be added only under `.task-list:not(.epm-issue-board) ...`. Do not change the baseline `.task-*`, `.story-*`, `.epic-*`, `.initiative-*`, or `.task-status` selectors that EPM intentionally shares. EPM rollup wrappers must keep `task-list epm-issue-board` on both all-project and selected-project render paths.

Filter card scope guard: compact stat-card overrides must be scoped as `.filters-strip .stat-card ...` only. Do not add or modify generic `.stat-card` rules for this change.

Analytics impact: no analytics event is needed. This is a visual density change to existing controls and does not add a new workflow action, trigger, or tracked user intent.

function cssRuleBody(source, selector) {
    const start = source.indexOf(`${selector} {`);
    assert.notStrictEqual(start, -1, `Expected CSS selector ${selector}`);
    const openBrace = source.indexOf('{', start);
    const closeBrace = source.indexOf('}', openBrace);
    assert.notStrictEqual(closeBrace, -1, `Expected CSS selector ${selector} to have a closing brace`);
    return source.slice(openBrace + 1, closeBrace);
}

function assertRuleIncludes(selector, declaration) {
    assert.ok(
        cssRuleBody(dashboardCssSource, selector).includes(declaration),
        `Expected ${selector} to include ${declaration}`
    );
}

function assertRuleExcludes(selector, declaration) {
    assert.equal(
        cssRuleBody(dashboardCssSource, selector).includes(declaration),
        false,
        `Expected ${selector} not to include compact declaration ${declaration}`
    );
}

const {
    activeHomeTokenConnection,
    appBaseUrl,

    selectedSprintName,

function populatedRollup(project) {
    return {
        projects: [
            {
                project,
                rollup: {
                    metadataOnly: false,
                    emptyRollup: false,
                    truncated: false,
                    truncatedQueries: [],
                    initiatives: {},
                    rootEpics: {
                        'EPM-100': {
                            issue: {
                                key: 'EPM-100',
                                summary: 'EPM compatibility epic with populated stories',
                                status: 'In Progress',
                                issueType: 'Epic',
                                assignee: 'Portfolio Owner',
                            },
                            stories: [
                                {
                                    key: 'EPM-101',
                                    summary: 'Visible EPM story keeps default issue-card spacing',
                                    status: 'In Progress',
                                    issueType: 'Story',
                                    priority: 'High',
                                    storyPoints: 3,
                                    assignee: 'EPM Engineer',
                                    updated: '2026-05-22T10:00:00.000Z',

                                {

                                    storyPoints: 5,
                                    assignee: 'EPM Engineer',
                                    updated: '2026-05-21T10:00:00.000Z',
                                },
                            ],
                        },
                    },
                    orphanStories: [],
                },
            },
        ],
        duplicates: {},
        truncated: false,
        fallback: false,

test('EPM issue board keeps default IssueCard density when ENG compact CSS is present', async ({ page }) => {
    await page.addInitScript((prefs) => {
        window.localStorage.setItem('jira_dashboard_ui_prefs_v1', JSON.stringify(prefs));
    }, {
        selectedView: 'epm',
        epmTab: 'active',
        epmSelectedProjectId: '',
        selectedSprint: selectedSprintId,
        sprintName: selectedSprintName,
    });
    await installDashboardFixture(page, {
        connection: activeHomeTokenConnection(),
        allProjectsRollup: populatedRollup,
    });

    await page.goto(`${appBaseUrl}/`, { waitUntil: 'networkidle' });

    const board = page.locator('.epm-issue-board');
    await expect(board).toHaveCount(1);
    await expect(board.locator('.epic-block')).toBeVisible();
    const firstStory = board.locator('.task-item[data-task-key="EPM-101"]');
    await expect(firstStory).toBeVisible();

    const styles = await firstStory.evaluate((node) => {
        const style = window.getComputedStyle(node);
        const titleStyle = window.getComputedStyle(node.querySelector('.task-title'));
        return {
            paddingTop: style.paddingTop,
            paddingRight: style.paddingRight,
            paddingBottom: style.paddingBottom,
            paddingLeft: style.paddingLeft,
            marginBottom: style.marginBottom,
            titleFontSize: titleStyle.fontSize,
        };
    });
    expect(styles.paddingTop).toBe('16px');
    expect(styles.paddingRight).toBe('20px');
    expect(styles.paddingBottom).toBe('16px');
    expect(styles.paddingLeft).toBe('20px');
    expect(styles.marginBottom).toBe('14.4px');
    expect(styles.titleFontSize).toBe('16.8px');

    await page.screenshot({ path: '/tmp/epm-issue-board-compact-scope.png', fullPage: false });
});
